    <script>
        // CRITICAL: Replace YOUR_GROQ_API_KEY with your actual key starting with gsk_
        const API_KEY = "YOUR_GROQ_API_KEY";

        async function process(target) {
            const text = document.getElementById('userInput').value;
            if(!text) { alert('Input cannot be empty!'); return; }

            document.getElementById('loader').classList.remove('hidden');
            document.getElementById('result').classList.add('hidden');

            try {
                const res = await fetch('https://api.groq.com/openai/v1/chat/completions', {
                    method: 'POST',
                    headers: {
                        'Authorization': `Bearer ${API_KEY}`,
                        'Content-Type': 'application/json'
                    },
                    body: JSON.stringify({
                        model: "llama-3.3-70b-versatile",
                        messages: [
                            {
                                role: "system", 
                                content: `You are a viral content strategist. Convert the following text into a high-engagement ${target}. Use power words, emojis, and clear formatting. Output the content ONLY.`
                            },
                            { role: "user", content: text }
                        ],
                        temperature: 0.8
                    })
                });

                const data = await res.json();
                document.getElementById('outputArea').innerText = data.choices[0].message.content;
                document.getElementById('loader').classList.add('hidden');
                document.getElementById('result').classList.remove('hidden');
            } catch (e) {
                alert('Connection Error. Check your API Key.');
                document.getElementById('loader').classList.add('hidden');
            }
        }

        function copyContent() {
            const text = document.getElementById('outputArea').innerText;
            navigator.clipboard.writeText(text);
            alert('Copied to clipboard!');
        }
    </script>
